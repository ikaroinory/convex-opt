\section*{8}

\begin{enumerate}[label=\alph*)]
    \item
        \begin{equation*}
            \partial|x|=
            \begin{cases}
                \{-1\}, &x<0, \\
                [-1,1], &x=0, \\
                \{1\}, &x>0.
            \end{cases}
        \end{equation*}
    \item
        \begin{equation*}
            \partial\max\left\{0, \frac{1}{2}\left(x^2-1\right)\right\}=
            \begin{cases}
                \{x\}, &x\in(-\infty, -1)\cup(1,+\infty), \\
                [-1,0], &x=-1, \\
                \{0\}, &x\in(-1,1), \\
                [0,1], &x=1.
            \end{cases}
        \end{equation*}
\end{enumerate}
